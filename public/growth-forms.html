开发者文档
飞书网页应用 + Bitable API 对接规范
适用接口: 飞书开放平台 v2 · 本文件可直接交给 Codex / 开发者使用
架构说明： 本系统采用「网页应用免登 + Bitable API 写入」方案。用户在飞书客户端内打开本页面，通过飞书 JSSDK 获取 code，后端用 code 换取 user_access_token，再调用 Bitable API 将表单数据写入对应多维表格。在非飞书环境下（普通浏览器），表格仍可正常使用并支持导出 CSV。
Step 1 — 前端免登流程（飞书客户端内）
/* 1. 检测飞书环境 */ const isFeishu = navigator.userAgent.includes('Lark') || navigator.userAgent.includes('Feishu'); /* 2. 在飞书内，调用 JSAPI 获取登录 code */ if (isFeishu && window.h5sdk) { window.h5sdk.ready(() => { window.h5sdk.requestAuthCode({ appId: 'YOUR_APP_ID', success(res) { const code = res.code; // 3. 将 code 发送给你的后端换取 user_access_token fetch('/api/feishu/auth', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ code }) }) .then(r => r.json()) .then(data => { window.__feishuToken = data.user_access_token; document.getElementById('feishu-badge').style.display = 'inline'; document.getElementById('user-info').textContent = data.user_name || '已登录'; }); }, fail(err) { console.error('飞书免登失败', err); } }); }); }
Step 2 — 后端代理接口（Node.js / Cloudflare Worker）
// /api/feishu/auth — 用 code 换 user_access_token app.post('/api/feishu/auth', async (req, res) => { const { code } = req.body; // 1. 获取 app_access_token const appTokenRes = await fetch('https://open.feishu.cn/open-apis/auth/v3/app_access_token/internal', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ app_id: process.env.FEISHU_APP_ID, app_secret: process.env.FEISHU_APP_SECRET }) }); const { app_access_token } = await appTokenRes.json(); // 2. 用 code 换 user_access_token const userTokenRes = await fetch('https://open.feishu.cn/open-apis/authen/v1/access_token', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${app_access_token}` }, body: JSON.stringify({ grant_type: 'authorization_code', code }) }); const userData = await userTokenRes.json(); res.json({ user_access_token: userData.data?.access_token, user_name: userData.data?.name }); });
Step 3 — 写入 Bitable API（前端调用示例）
/** * 将表格数据提交到飞书多维表格 * @param {string} tableId - 多维表格的 table_id（tblXXX） * @param {object} fields - 字段键值对（参照下方字段映射表） */ async function submitRecord(tableId, fields) { const APP_TOKEN = 'YOUR_BITABLE_APP_TOKEN'; // 从多维表格 URL /base/ 后获取 const token = window.__feishuToken || await getTenantToken(); // 非登录态降级用 tenant_token const res = await fetch( `https://open.feishu.cn/open-apis/bitable/v1/apps/${APP_TOKEN}/tables/${tableId}/records`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ fields }) } ); return res.json(); } // 示例：提交 B4 素材测试一条记录 submitRecord('TBL_B4_ID', { creative_id: document.getElementById('cr-id').value, script_type: document.getElementById('script-type').value, thumb_stop_rate: parseFloat(document.getElementById('ts-rate').value), hold_rate: parseFloat(document.getElementById('hold-rate').value), ctr: parseFloat(document.getElementById('ctr').value), cvr: parseFloat(document.getElementById('cvr').value), action: document.getElementById('action').value });
